Up The Interfaces With Security Zones
--------------------------------------------------------------

PERIMETER-FW2(config)#interface GigabitEthernet1/3

PERIMETER-FW2(config-if)#nameif INSIDE

Static + OSPF Routing
--------------------------------------------------------------
PERIMETER-FW2(config)#route OUTSIDE 0.0.0.0 0.0.0.0 197.200.100.5
PERIMETER-FW2(config)#route OUTSIDE2 0.0.0.0 0.0.0.0 105.100.50.5

PERIMETER-FW2(config)#router ospf 35
PERIMETER-FW2(config-router)#router-id 1.1.9.9
PERIMETER-FW2(config-router)#network 105.100.50.4 255.255.255.252 area 0
PERIMETER-FW2(config-router)#network 197.200.100.4 255.255.255.252 area 0
PERIMETER-FW2(config-router)#network 10.2.2.4 255.255.255.252 area 0
PERIMETER-FW2(config-router)#network 10.2.2.12 255.255.255.252 area 0
PERIMETER-FW2(config-router)#ex
